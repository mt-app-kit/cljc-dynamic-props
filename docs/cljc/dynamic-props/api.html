(defn get-props
  [component-id & [default-props]]
  (if-let [dynamic-props (common-state/get-state :dynamic-props component-id)]
          (-> default-props (map/deep-merge dynamic-props))

import-props
(defn import-props
  [component-id & [component-props default-props]]
  (if-let [dynamic-props (common-state/get-state :dynamic-props component-id)]
          (-> component-props (map/deep-merge dynamic-props))
          (or component-props {})))
description
Merges the dynamic properties of the component (stored in the common state atom) onto the given property map.

(defn update-props!
  [component-id f & params]
  (letfn [(f0 [%] (apply f % params))]
         (common-state/update-state! :dynamic-props component-id f0)))

(defn merge-props!
